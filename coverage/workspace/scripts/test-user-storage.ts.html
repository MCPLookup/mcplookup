      const healthResult = await userStorage.healthCheck();

          console.log(`   Users: ${healthResult.details.userCount}`);

        id: userId,
        email: 'test@example.com',
        name: 'Test User',
        image: 'https://example.com/avatar.jpg',
        provider: 'github',
        provider_id: 'github123',
        role: 'user',
        created_at: new Date().toISOString(),
        updated_at: new Date().toISOString(),
        email_verified: true,
        is_active: true,
        preferences: {
          theme: 'dark',
          notifications: true,
          newsletter: false
        }
      };
 
      const storeResult = await userStorage.storeUser(userId, testUser);
      if (storeResult.success) {
        console.log('   ‚úÖ User created successfully');
      } else {
        console.log(`   ‚ùå Failed to create user: ${storeResult.error}`);
        continue;
      }
 
      // Test user retrieval
      console.log('\nüîç Testing user retrieval...');
      
      // Get by ID

      if (getUserResult.success && getUserResult.data) {
        console.log('   ‚úÖ Get user by ID: Success');
        console.log(`   üìß Email: ${getUserResult.data.email}`);
        console.log(`   üë§ Name: ${getUserResult.data.name}`);
        console.log(`   üîê Provider: ${getUserResult.data.provider}`);
      } else {
        console.log('   ‚ùå Get user by ID: Failed');
      }
 
      // Get by email
      const getUserByEmailResult = await userStorage.getUserByEmail('test@example.com');
      if (getUserByEmailResult.success && getUserByEmailResult.data) {
        console.log('   ‚úÖ Get user by email: Success');
      } else {
        console.log('   ‚ùå Get user by email: Failed');
      }

      const getUserByProviderResult = await userStorage.getUserByProvider('github', 'github123');

        console.log('   ‚úÖ Get user by provider: Success');
      } else {
        console.log('   ‚ùå Get user by provider: Failed');

        name: 'Updated Test User',

        email_verified: false

        console.log('   ‚úÖ User updated successfully');
        
        // Verify update
        const updatedUserResult = await userStorage.getUser(userId);
        if (updatedUserResult.success && updatedUserResult.data) {
          console.log(`   üìù Updated name: ${updatedUserResult.data.name}`);
          console.log(`   üîë Updated role: ${updatedUserResult.data.role}`);
          console.log(`   üìß Email verified: ${updatedUserResult.data.email_verified}`);
        }
      } else {
        console.log(`   ‚ùå Failed to update user: ${updateResult.error}`);
      }

      console.log('\nüîê Testing session management...');
      const sessionId = randomUUID();

        id: sessionId,
        user_id: userId,
        token: 'test-session-token-123',
        expires_at: new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString(), // 24 hours

        ip_address: '127.0.0.1',
        user_agent: 'Test User Agent'

          console.log('   ‚úÖ Session retrieved successfully');
          console.log(`   üé´ Token: ${getSessionResult.data.token.substring(0, 20)}...`);
        }
      } else {
        console.log(`   ‚ùå Failed to create session: ${storeSessionResult.error}`);
      }
 
      // Test registration tracking
      console.log('\nüìù Testing registration tracking...');
      const registrationId = randomUUID();
      const testRegistration: UserRegistration = {
        id: registrationId,
        user_id: userId,
        domain: 'test-example.com',
        challenge_id: 'challenge-123',
        created_at: new Date().toISOString(),
        status: 'pending'
      };
 
      const storeRegResult = await userStorage.storeRegistration(registrationId, testRegistration);
      if (storeRegResult.success) {
        console.log('   ‚úÖ Registration created successfully');
        
        // Test registration retrieval
        const getRegResult = await userStorage.getRegistrationsByUser(userId);
        if (getRegResult.success && getRegResult.data.items.length > 0) {
          console.log('   ‚úÖ Registration retrieved successfully');
          console.log(`   üåê Domain: ${getRegResult.data.items[0].domain}`);
          console.log(`   üìä Status: ${getRegResult.data.items[0].status}`);
        }
      } else {
        console.log(`   ‚ùå Failed to create registration: ${storeRegResult.error}`);
      }
 
      // Test user search and filtering
      console.log('\nüîç Testing user search and filtering...');
      const getAllResult = await userStorage.getAllUsers({ limit: 10 });
      if (getAllResult.success) {
        console.log(`   ‚úÖ Found ${getAllResult.data.total} users`);
        console.log(`   üìÑ Page size: ${getAllResult.data.items.length}`);

        console.log(`   ‚úÖ Search found ${searchResult.data.total} matching users`);

      console.log('\nüìä Testing user statistics...');
      const statsResult = await userStorage.getStats();
      if (statsResult.success) {
        console.log('   ‚úÖ Statistics retrieved successfully');
        console.log(`   üë• Total users: ${statsResult.data.totalUsers}`);
        console.log(`   ‚úÖ Active users: ${statsResult.data.activeUsers}`);
        console.log(`   üìß Verified users: ${statsResult.data.verifiedUsers}`);
